📚 Block 1 — Edge Case Theory & Pitfalls

⏱️ 3 Hours
First principle: Judges craft inputs specifically to break naive assumptions. If your solution passes the sample but fails the judge, you hit an edge case. Know them all before coding.

Empty, single, all-same, increasing/decreasing, overflow, negatives, duplicates, circular, boundary indices.

🔴 Empty (n=0): Add guard before any access.

🟡 Single (n=1): Init with arr[0], never 0 or INT_MIN.

🟠 Overflow: sum ≤ 10^14 — use long long.

🟢 Negatives: Sliding window breaks — must handle separately.

🔵 Circular: Use i % n or complement trick: total - min_subarray.

⚪ Boundary: Loop ends at i < n, not i ≤ n.

Off-by-one, index OOB, integer overflow, wrong sorted assumptions.

// OFF-BY-ONE: prefix query
// WRONG: prefix[r] - prefix[l-1]
// RIGHT: prefix[r+1] - prefix[l]   (0-indexed)

// OVERFLOW: use long long for sums
long long sum = (long long)a + b;

// INDEX OOB: cache n, don't use .size()-1 unsigned
int n = arr.size();
for(int i = 0; i < n-1; i++) ...  // safe

// UNSIGNED: arr.size()-1 when size==0 → HUGE number
// Always cache: int n = arr.size();

🟢 Block 2 — Level 1: Warm-Up (P1–P4)

⏱️ 3 Hours

Build O(n) prefix array → answer each query in O(1). Handles negatives perfectly.

// Build: O(n)  Query: O(1)
vector<long long> prefix(n+1, 0);
for(int i=0; i<n; i++) prefix[i+1] = prefix[i] + A[i];
// Query sum(l,r):
long long ans = prefix[r+1] - prefix[l];

n*(n+1)/2 - sum(array). Edge: missing 0, missing n, n=1.

int missingNumber(vector<int>& nums) {
    long long n = nums.size();
    long long expected = n*(n+1)/2;
    long long actual = 0;
    for(int x: nums) actual += x;
    return expected - actual;
}

Single write pointer advances only on non-zero. Then fill tail with zeros.

void moveZeroes(vector<int>& nums) {
    int w = 0;
    for(int x: nums) if(x) nums[w++] = x;
    while(w < nums.size()) nums[w++] = 0;
}

Count drop points A[i] > A[(i+1)%n]. At most 1 drop → valid rotation.

bool check(vector<int>& nums) {
    int n=nums.size(), drops=0;
    for(int i=0;i<n;i++)
        if(nums[i]>nums[(i+1)%n]) drops++;
    return drops <= 1;
}

🟡 Block 3 — Level 2: Core Patterns (P5–P10)

⏱️ 5 Hours

⚠️ Only works for non-negative values. Window shrinks when sum > k.

int solve(vector<int>& A, int k) {
    int l=0, ans=0; long long sum=0;
    for(int r=0; r<A.size(); r++) {
        sum += A[r];
        while(sum > k) sum -= A[l++];
        ans = max(ans, r-l+1);
    }
    return ans;
}

🔑 Sliding window FAILS with negatives. map[0]=1 is critical — don't forget!

int subarraySum(vector<int>& nums, int k) {
    unordered_map<long long,int> freq;
    freq[0] = 1;  // ← CRITICAL: empty prefix
    long long prefix=0; int cnt=0;
    for(int x: nums) {
        prefix += x;
        cnt += freq[prefix - k];
        freq[prefix]++;
    }
    return cnt;
}

dp[i] = max(dp[i-2]+A[i], dp[i-1]). Compress to 2 variables.

int rob(vector<int>& A) {
    int prev2=0, prev1=0;
    for(int x: A) {
        int curr = max(prev1, prev2+x);
        prev2=prev1; prev1=curr;
    }
    return prev1;
}

Shrink from left whenever sum ≥ s. Return 0 if no valid subarray found.

int minSubArrayLen(int s, vector<int>& A) {
    int l=0, ans=INT_MAX; long long sum=0;
    for(int r=0; r<A.size(); r++) {
        sum+=A[r];
        while(sum>=s) { ans=min(ans,r-l+1); sum-=A[l++]; }
    }
    return ans==INT_MAX ? 0 : ans;
}

Forward pass: prefix products. Backward pass: multiply by suffix. O(1) extra space.

vector<int> productExceptSelf(vector<int>& A) {
    int n=A.size(); vector<int> res(n,1);
    int left=1;
    for(int i=0;i<n;i++){ res[i]=left; left*=A[i]; }
    int right=1;
    for(int i=n-1;i>=0;i--){ res[i]*=right; right*=A[i]; }
    return res;
}

O(1) space: whichever side is shorter, its max determines water. Move inward.

int trap(vector<int>& h) {
    int l=0, r=h.size()-1, lm=0, rm=0, w=0;
    while(l<r) {
        if(h[l]<=h[r]) { lm=max(lm,h[l]); w+=lm-h[l++]; }
        else            { rm=max(rm,h[r]); w+=rm-h[r--]; }
    }
    return w;
}

🔴 Block 4 — Level 3: Advanced (P11–P15)

⏱️ 5 Hours

Sort by |abs| desc. Flip negatives greedily. If k odd remains, flip smallest |abs|.

int largestSumAfterKNegations(vector<int>& A, int k) {
    sort(A.begin(),A.end(),[](int a,int b){return abs(a)>abs(b);});
    for(int i=0;i<A.size()&&k;i++) if(A[i]<0){A[i]=-A[i];k--;}
    if(k%2) A.back()=-A.back();
    return accumulate(A.begin(),A.end(),0);
}

Build decreasing stack (left candidates), sweep right-to-left to maximise j-i.

int maxWidthRamp(vector<int>& A) {
    int n=A.size(), ans=0; stack<int> st;
    for(int i=0;i<n;i++) if(st.empty()||A[st.top()]>A[i]) st.push(i);
    for(int j=n-1;j>=0;j--)
        while(!st.empty()&&A[st.top()]<=A[j])
            { ans=max(ans,j-st.top()); st.pop(); }
    return ans;
}

Negatives break sliding window. Use prefix sum + monotone deque. O(n).

int shortestSubarray(vector<int>& A, int k) {
    int n=A.size(), ans=INT_MAX;
    vector<long long> pre(n+1,0);
    for(int i=0;i<n;i++) pre[i+1]=pre[i]+A[i];
    deque<int> dq;
    for(int r=0;r<=n;r++) {
        while(!dq.empty()&&pre[r]-pre[dq.front()]>=k)
            { ans=min(ans,r-dq.front()); dq.pop_front(); }
        while(!dq.empty()&&pre[dq.back()]>=pre[r]) dq.pop_back();
        dq.push_back(r);
    }
    return ans==INT_MAX?-1:ans;
}

Each element is the min for exactly one maximal range → use prev/next-smaller stacks.

long long maxScore(vector<int>& A) {
    int n=A.size(); long long ans=0;
    vector<int> ps(n,-1), ns(n,n); stack<int> st;
    for(int i=0;i<n;i++){ while(!st.empty()&&A[st.top()]>=A[i])st.pop(); ps[i]=st.empty()?-1:st.top(); st.push(i); }
    while(!st.empty())st.pop();
    for(int i=n-1;i>=0;i--){ while(!st.empty()&&A[st.top()]>=A[i])st.pop(); ns[i]=st.empty()?n:st.top(); st.push(i); }
    for(int i=0;i<n;i++) ans=max(ans,(long long)A[i]*(ns[i]-ps[i]-1));
    return ans;
}

Normalize negative remainders: ((prefix%k)+k)%k. C++ % can be negative.

int subarraysDivByK(vector<int>& A, int k) {
    unordered_map<int,int> freq; freq[0]=1;
    int prefix=0, cnt=0;
    for(int x: A) {
        prefix += x;
        int rem = ((prefix%k)+k)%k;  // normalize negatives
        cnt += freq[rem];
        freq[rem]++;
    }
    return cnt;
}
