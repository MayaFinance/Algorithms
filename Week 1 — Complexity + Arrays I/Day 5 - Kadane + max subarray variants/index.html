📚 Block 1 — Theory & Mathematical Foundations

⏱️ 3 Hours
Jay Kadane's Insight (1984): Instead of re-evaluating all possible start/end pairs, track just one number —the best possible sum that MUST end at the current position. If extending from the past helps, do it. If not, start fresh.

Formalize the local/global relationship.

f(i) = max(A[i], f(i-1) + A[i])

If f(i-1) is negative: reset — start fresh at A[i].
If f(i-1) is positive: extend — adding it only helps.
Global max = max over all f(i).

int maxSubArray(vector<int>& nums) {
    int curr = nums[0], best = nums[0];
    for (int i = 1; i < nums.size(); i++) {
        curr = max(nums[i], curr + nums[i]);
        best = max(best, curr);
    }
    return best;
}

🔄 Block 2 — Core Variants

⏱️ 4 Hours

Dual case: non-wrap (standard Kadane) vs wrap (total - min_subarray).

int maxSubarraySumCircular(vector<int>& nums) {
    int total=0, max_s=nums[0], curr_max=nums[0];
    int min_s=nums[0], curr_min=nums[0];
    for (int i=1; i<nums.size(); i++) {
        curr_max = max(nums[i], curr_max + nums[i]);
        max_s = max(max_s, curr_max);
        curr_min = min(nums[i], curr_min + nums[i]);
        min_s = min(min_s, curr_min);
        total += nums[i];
    }
    total += nums[0];
    // Guard: all-negative case
    return (total == min_s) ? max_s : max(max_s, total - min_s);
}

Track both curr_max and curr_min — a negative flips to become the new max.

int maxProduct(vector<int>& nums) {
    int cmax=nums[0], cmin=nums[0], res=nums[0];
    for (int i=1; i<nums.size(); i++) {
        int x = nums[i];
        int tmax = max({x, cmax*x, cmin*x});
        int tmin = min({x, cmax*x, cmin*x});
        cmax = tmax; cmin = tmin;
        res = max(res, cmax);
    }
    return res;
}

Classic Kadane in disguise: track running min price + max profit.

int maxProfit(vector<int>& prices) {
    int min_p = INT_MAX, profit = 0;
    for (int p : prices) {
        min_p  = min(min_p, p);
        profit = max(profit, p - min_p);
    }
    return profit;
}

Two Kadane passes: left[i] + right[i] bridged at each deletion point k.

int maximumSum(vector<int>& arr) {
    int n = arr.size();
    vector<int> L(n), R(n);
    L[0] = arr[0];
    for (int i=1; i<n; i++)
        L[i] = max(arr[i], L[i-1]+arr[i]);
    R[n-1] = arr[n-1];
    for (int i=n-2; i>=0; i--)
        R[i] = max(arr[i], R[i+1]+arr[i]);
    int res = *max_element(L.begin(), L.end());
    for (int k=1; k<n-1; k++)
        res = max(res, L[k-1] + R[k+1]);
    return res;
}

🔥 Block 3 — Advanced Variants

⏱️ 4 Hours

Track both ends: max Kadane and min Kadane simultaneously.

int maxAbsoluteSum(vector<int>& nums) {
    int f=0, g=0, ans=0;
    for (int x : nums) {
        f = max(f,0) + x;      // max ending here
        g = min(g,0) + x;      // min ending here
        ans = max({ans, f, abs(g)});
    }
    return ans;
}

Row-compress each column pair into 1D → run Kadane. O(C² × R).

int maxSumRectangle(vector<vector<int>>& M) {
    int R=M.size(), C=M[0].size(), best=INT_MIN;
    for (int l=0; l<C; l++) {
        vector<int> rs(R, 0);
        for (int r=l; r<C; r++) {
            for (int i=0; i<R; i++) rs[i] += M[i][r];
            // Kadane on rs[]
            int c=rs[0], b=rs[0];
            for (int i=1; i<R; i++) {
                c = max(rs[i], c+rs[i]);
                b = max(b, c);
            }
            best = max(best, b);
        }
    }
    return best;
}

Kadane maximizes. For exact targets, use Prefix Sum + HashMap instead.

int subarraySum(vector<int>& nums, int k) {
    unordered_map<int,int> mp; mp[0]=1;
    int sum=0, cnt=0;
    for (int x : nums) {
        sum += x;
        cnt += mp[sum - k];
        mp[sum]++;
    }
    return cnt;
}

🌳 Block 4 — Segment Trees for Range Queries

⏱️ 3 Hours
When does standard Kadane fail? When you receive multiple range queries [l, r] asking for max subarray in that range. Kadane does O(n) per query. A Segment Tree answers in O(log n).

Each segment tree node must store 4 values:

  • total — Sum of all elements in range
  • max_prefix — Largest sum starting from left of range
  • max_suffix — Largest sum ending at right of range
  • max_sub — Best subarray inside the range

Merge: max_sub = max(left.max_sub, right.max_sub, left.max_suffix + right.max_prefix)
