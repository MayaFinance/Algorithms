⚡ Block 1 — 1D Mathematical Foundations

⏱️ 3 Hours
The Discrete Derivative: Just as a continuous function is reconstructed from its derivative via integration, a discrete sequence is reconstructed from its difference array via prefix summation.

Understand the initialization and relation to Prefix Sums.

The Difference array captures the "delta" or relative change at each index:

D[i] = A[i] - A[i-1]

By definition, A[k] = Σ D[i] (from i=0 to k). This duality makes range updates O(1). If we add `v` at `D[l]`, ALL elements from `l` onward will increase by `v` during the prefix sum pass!

Adding +v to Range [L, R] using only 2 operations.

vector<int> getModifiedArray(int n, vector<vector<int>>& updates) {
    // Crucial: Allocate size N + 1 to prevent Out-Of-Bounds on R + 1
    vector<int> diff(n + 1, 0); 
    
    // 1. O(K) Batch Processing
    for(auto& u : updates) {
        int l = u[0], r = u[1], val = u[2];
        diff[l] += val;             // Start increasing here
        diff[r + 1] -= val;         // Stop increasing past this
    }
    
    // 2. O(N) Reconstruction 
    vector<int> res(n);
    int current = 0;
    for(int i = 0; i < n; i++) {
        current += diff[i];
        res[i] = current;
    }
    return res;
}

Converting 1-indexed problem domains to 0-indexed memory spaces.

vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
    // Array of size n+2 handles 1-based indexing offsets smoothly
    vector<int> diff(n + 2, 0);
    
    for (auto& b : bookings) {
        int first = b[0], last = b[1], seats = b[2];
        diff[first] += seats;     // First mapped cleanly
        diff[last + 1] -= seats;  // Drop safely into buffer
    }
    
    vector<int> result(n);
    int current_seats = 0;
    for (int i = 1; i <= n; i++) { // Reconstruct starting at 1
        current_seats += diff[i];
        result[i - 1] = current_seats; // Store correctly in 0-based result
    }
    return result;
}

Advanced: Maximize sum of queries by rearranging initial elements.

Core Idea: Identify which indices are queried the most. Place the largest numbers from the array into those high-frequency indices.

long long maxSum(vector<int>& arr, vector<pair<int, int>>& queries) {
    int n = arr.size();
    vector<int> diff(n + 1, 0);
    
    // 1. Tally query frequencies using Difference Array
    for(auto& q : queries) {
        diff[q.first - 1]++;
        diff[q.second]--;
    }
    
    // 2. Reconstruct frequencies
    vector<int> freq(n);
    int curr = 0;
    for(int i = 0; i < n; i++) {
        curr += diff[i];
        freq[i] = curr;
    }
    
    // 3. Sort both frequencies and array
    sort(freq.begin(), freq.end());
    sort(arr.begin(), arr.end());
    
    // 4. Dot Product (Greedy)
    long long total = 0;
    for(int i = 0; i < n; i++) {
        total += 1LL * freq[i] * arr[i];
    }
    return total;
}

Standard 1D difference array simulation to track maximum capacity usage over time.

bool carPooling(vector<vector<int>>& trips, int capacity) {
    int maxLocation = 0;
    for (auto& t : trips) maxLocation = max(maxLocation, t[2]);
    vector<int> diff(maxLocation + 1, 0);
    for (auto& t : trips) { diff[t[1]] += t[0]; diff[t[2]] -= t[0]; }
    
    int current_passengers = 0;
    for (int i = 0; i <= maxLocation; i++) {
        current_passengers += diff[i];
        if (current_passengers > capacity) return false;
    }
    return true;
}

Use a TreeMap (`std::map` in C++) as a Difference Array to handle implicitly large coordinate spaces.

class MyCalendarTwo {
    map<int, int> diff;
public:
    bool book(int start, int end) {
        diff[start]++;
        diff[end]--;
        int active = 0;
        for (auto& [time, delta] : diff) {
            active += delta;
            if (active >= 3) {
                diff[start]--; diff[end]++;
                return false;
            }
        }
        return true;
    }
};

Sweep-line / Difference Array with mapping to handle intervals and merging properties (colors) on a number line.

vector<vector<long long>> splitPainting(vector<vector<int>>& segments) {
    map<int, long long> diff;
    for (auto& seg : segments) {
        diff[seg[0]] += seg[2]; diff[seg[1]] -= seg[2];
    }
    vector<vector<long long>> res;
    long long color_sum = 0;
    int last_pos = -1;
    for (auto& [pos, delta] : diff) {
        if (color_sum > 0 && last_pos != -1) res.push_back({last_pos, pos, color_sum});
        color_sum += delta;
        last_pos = pos;
    }
    return res;
}

Using `diff[r][c] += v` logic scaling over submatrices.

To add `v` into a rectangle from (r1, c1) to (r2, c2):

diff[r1][c1] += v
diff[r2+1][c1] -= v
diff[r1][c2+1] -= v
diff[r2+1][c2+1] += v
